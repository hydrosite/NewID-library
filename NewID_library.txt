local NewID = {}
NewID.Gui = {}

function NewID.Gui:new()
    -- Create the loading screen
    local loadingScreen = Instance.new("ScreenGui")
    loadingScreen.Name = "LoadingScreen"
    loadingScreen.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

    -- Background color (soft gradient)
    loadingScreen.BackgroundColor3 = Color3.fromRGB(40, 44, 52)

    -- Create the ball
    local ball = Instance.new("Frame")
    ball.Size = UDim2.new(0, 70, 0, 70) -- Larger size for better visibility
    ball.Position = UDim2.new(1, -35, 0.5, 0) -- Start from the right side
    ball.BackgroundColor3 = Color3.fromRGB(25, 25, 25) -- Darker ball color
    ball.AnchorPoint = Vector2.new(0.5, 0.5)
    ball.Parent = loadingScreen

    -- Create a UICorner to make the ball circular
    local uicorner = Instance.new("UICorner")
    uicorner.CornerRadius = UDim.new(1, 0) -- Perfectly circular ball
    uicorner.Parent = ball

    -- Glowing effect around the ball
    local uiGradient = Instance.new("UIGradient")
    uiGradient.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(50, 50, 50)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(0, 0, 0))
    }
    uiGradient.Rotation = 90
    uiGradient.Parent = ball

    -- Add the letter "N" to the ball
    local letterN = Instance.new("TextLabel")
    letterN.Size = UDim2.new(1, 0, 1, 0)
    letterN.Position = UDim2.new(0, 0, 0, 0)
    letterN.Text = "N"
    letterN.TextColor3 = Color3.fromRGB(240, 128, 128) -- Softer, more modern red
    letterN.Font = Enum.Font.GothamBlack -- Sleek, modern font
    letterN.TextScaled = true
    letterN.BackgroundTransparency = 1
    letterN.Parent = ball

    -- Create the loading text for "New"
    local loadingTextNew = Instance.new("TextLabel")
    loadingTextNew.Size = UDim2.new(0, 120, 0, 60)
    loadingTextNew.Position = UDim2.new(0.5, -60, 0.5, -90) -- Slightly above the ball
    loadingTextNew.Text = "New"
    loadingTextNew.TextColor3 = Color3.fromRGB(255, 255, 255) -- Clean white text
    loadingTextNew.Font = Enum.Font.GothamBlack
    loadingTextNew.TextScaled = true
    loadingTextNew.BackgroundTransparency = 1
    loadingTextNew.Visible = false -- Hidden initially
    loadingTextNew.Parent = loadingScreen

    -- Create the loading text for "ID"
    local loadingTextID = Instance.new("TextLabel")
    loadingTextID.Size = UDim2.new(0, 120, 0, 60)
    loadingTextID.Position = UDim2.new(0.5, -60, 0.5, 20) -- Below the "New" text
    loadingTextID.Text = "ID"
    loadingTextID.TextColor3 = Color3.fromRGB(135, 206, 250) -- Light blue for a modern touch
    loadingTextID.Font = Enum.Font.GothamBlack
    loadingTextID.TextScaled = true
    loadingTextID.BackgroundTransparency = 1
    loadingTextID.Visible = false -- Hidden initially
    loadingTextID.Parent = loadingScreen

    -- Ball animation to slide smoothly to the center
    local targetPosition = UDim2.new(0.5, -35, 0.5, 0)
    ball:TweenPosition(targetPosition, "Out", "Quad", 1.2, true) -- Smoother animation

    wait(1.2) -- Wait for the ball to reach the center

    -- Transition to text by shrinking the ball
    ball:TweenSize(UDim2.new(0, 0, 0, 0), "Out", "Quad", 0.7, true)

    -- Wait for the ball to shrink
    wait(0.7)

    -- Show the loading text after the ball animation
    loadingTextNew.Visible = true
    loadingTextID.Visible = true

    -- Smooth fade-in and expand the text
    loadingTextNew:TweenSize(UDim2.new(0, 400, 0, 100), "Out", "Quad", 0.5, true)
    loadingTextNew:TweenPosition(UDim2.new(0.5, -200, 0.5, -70), "Out", "Quad", 0.5, true)

    loadingTextID:TweenSize(UDim2.new(0, 400, 0, 100), "Out", "Quad", 0.5, true)
    loadingTextID:TweenPosition(UDim2.new(0.5, -200, 0.5, 30), "Out", "Quad", 0.5, true)

    wait(1) -- Wait for the text animation to finish

    -- Remove the loading screen after the animation
    loadingScreen:Destroy()
end

return NewID
